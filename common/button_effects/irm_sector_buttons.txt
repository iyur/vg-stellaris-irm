# Custom sector ui effect buttons

# Sector type
irm_bt_sector_type = {
	potential = { exists = event_target:irm_ui_sector }
	effect = {
		hidden_effect = {
			event_target:irm_ui_sector = {
				if = {
					limit = { has_sector_type = core_sector }
						custom_tooltip = "irm.ui.sector.type.core.tt"
				}
				else_if = {
					limit = { has_sector_type = normal_sector }
						custom_tooltip = "irm.ui.sector.type.normal.tt"
				}
				# todo: star sector
			}
		}
	}
}

irm_bt_sector_energy = {
	potential = { exists = event_target:irm_ui_sector }
	effect = {
		hidden_effect = {
			event_target:irm_ui_sector = {
				custom_tooltip = "irm.ui.sector.res.energy.tt"
				custom_tooltip = "irm.ui.sector.res.tt.inc"
				if = {
					limit = { check_variable = { which = irm_energy_base_inc value > 0 } }
					custom_tooltip = "irm.ui.sector.res.energy.tt.base.inc"
				}
				if = {
					limit = { check_variable = { which = irm_energy_station_inc value > 0 } }
					custom_tooltip = "irm.ui.sector.res.energy.tt.station.inc"
				}
				if = {
					limit = { check_variable = { which = irm_energy_planet_inc value > 0 } }
					custom_tooltip = "irm.ui.sector.res.energy.tt.planet.inc"
				}
				custom_tooltip = "irm.ui.sector.res.tt.exp"
				if = {
					limit = { check_variable = { which = irm_energy_station_exp value > 0 } }
					custom_tooltip = "irm.ui.sector.res.energy.tt.station.exp"
				}
				if = {
					limit = { check_variable = { which = irm_energy_starbase_exp value > 0 } }
					custom_tooltip = "irm.ui.sector.res.energy.tt.starbase.exp"
				}
				if = {
					limit = { check_variable = { which = irm_energy_planet_exp value > 0 } }
					custom_tooltip = "irm.ui.sector.res.energy.tt.planet.exp"
				}
				custom_tooltip = "irm.ui.sector.res.energy.tte"
			}
		}
	}	
}
